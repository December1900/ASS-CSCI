\item 
BPMN for updating username with cumulative effects:

\includegraphics[scale=0.3]{Figure4-3.jpeg}

\begin{enumerate}
      \item Objects of interests: Application a, Username u, Outcome o, Status s.
      \item States:
          \begin{itemize}
              \item[·] Application rejected or approved - rejected(a), approved(a)
              \item[·] Username check approved or rejected, update successed- approved(u), rejected(u), successed(u)
          \end{itemize}
      \item Relationships between objects:
          \begin{itemize}
              \item[·] Application investigated with an outcome - investigated(a,o)
              \item[·] Username stored with an status - stored(u,s)
              \item[·] Username updated with an stauts - updated(u,s)
              \item[·] Usernmae received with an status - received(u,s)
          \end{itemize}
      \item Cumulative effects of tasks:
          \begin{enumerate}
            \item[1.]Reject Application: \textcolor{red}{investigated(a,o) $\land$ rejected(a)}
            \item[2.] Store Username: \textcolor{red}{investigated(a,o) $\land$ approved(a) $\land$ approved(u) $\land$ stored(u,s)}
            \item[3.] Reject Update Username: \textcolor{red}{investigated(a,o) $\land$ approved(a) $\land$ approved)(u) $\land$ interrupted(u) $\land$ rejected(u)}
            \item[4.] Update Success: \textcolor{red}{investigated(a,o) $\land$ approved(a) $\land$ approved(u) $\land$ stored(u,s) $\land$ successed(u)}
            \item[5.] Update Username: \textcolor{red}{investigated(a,o) $\land$ approved(a) $\land$ approved(u) $\land$ stored(u,s) $\land$ successed(u) $\land$ updated(u,s)}
          \end{enumerate}
\end{enumerate}

              \item[·] Password tested with an status - tested(p,s)
              \item[·] Password encrypted with an status - encrypted(p,s) 
              \item[·] Password updated with an status - updated(p,s)
              \item[·] 
          \end{itemize}
      \item Cumulative effects of tasks:
          \begin{enumerate}
            \item[1.] Set Password: \textcolor{red}{approved(a) $\land$ set(p,o)}
            \item[2.] Acquire Password: \textcolor{red}{approved(a) $\land$ set(p,o) $\land$ acquired(p,o)}
            \item[3.] Reject Update Password: \textcolor{red}{approved(a) $\land$ set(p,o) $\land$ acquired(p,o) $\land$ unavailabled(p) $\land$ rejected(p)}

            \item[7.] Update Password : \textcolor{red}{approved(a) $\land$ set(p,o) $\land$ acquired(p,o) $\land$ (availabled(p) $\land$ tested(p,s) $\land$ (availabled(p) $\land$ encrypted(p,s))) $\land$ successed(p) $\land$ updated(p,s)}

            
        \end{enumerate}
